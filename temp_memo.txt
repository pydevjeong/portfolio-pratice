  }),
};

const components = {
  Heading: {
    variants: {
      "section-title": {
        textDecoration: "underline",
        fontSize: 20,

        textDecorationColor: "#525252",
        textDecorationThickness: 4,

        marginBottom: 4,
      },
    },
  },
  Link: {
    baseStyle: (props) => ({
      color: mode("#3d7aed", "#ff63c3")(props),

    }),
  },
};

  heading: "'M PLUS Rounded 1c'",
};

  glassTeal: "#88ccaa",

  initialColorMode: "dark",

};

  config,
  styles,
  components,
  colors,
  fonts,
});
